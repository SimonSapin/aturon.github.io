fn find_mut_with(&'a mut self, f: |&K| -> Ordering) -> Option<&'a mut V>

Example

let mut t = collections::treemap::TreeMap::new();
t.insert("Content-Type", "application/xml");

let new_ua = "Safari/156.0";

   None => fail!(),
