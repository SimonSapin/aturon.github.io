macro_rules! green_start( ($f:ident) => (
    mod __start {
        extern crate green;
        extern crate rustuv;

        #[start]
        fn start(argc: int, argv: *const *const u8) -> int {
            green::start(argc, argv, rustuv::event_loop, super::$f)
        }
    }
) )

A helper macro for booting a program with libgreen

Example

#![feature(phase)] #[phase(plugin)] extern crate green; green_start!(main) fn main() { // running with libgreen }
