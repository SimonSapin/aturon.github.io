Function url::encodeExperimental[src]

pub fn encode<T: BytesContainer>(container: T) -> String

Encodes a URI by replacing reserved characters with percent-encoded character sequences.

This function is compliant with RFC 3986.

Example

extern crate url; fn main() { use url::encode; let url = encode("https://example.com/Rust (programming language)"); println!("{}", url); // https://example.com/Rust%20(programming%20language) }
use url::encode;

let url = encode("https://example.com/Rust (programming language)");
